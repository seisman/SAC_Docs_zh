\SACCMD{plotcor}
\label{spe:plotcor}

\SACTitle{概要}
绘制相关函数

\SACTitle{语法}
\begin{SACSTX}
P!LOT!COR [X!LIM! v|ON|OFF]
\end{SACSTX}

\SACTitle{输入}
\begin{description}
\item [XLIM v] 打开X轴范围选项，并设置上限为v秒，下限一直为0
\item [XLIM ON] 打开X轴范围选项，并使用原先设置的上限
\item [XLIM OFF] 关闭X轴范围选项，即绘制全部相关函数
\end{description}

\SACTitle{缺省值}
\begin{SACDFT}
plotcor xlim off
\end{SACDFT}

\SACTitle{相关命令}
\nameref{spe:cor}
