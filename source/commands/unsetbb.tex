\SACCMD{unsetbb}
\label{cmd:unsetbb}

\SACTitle{概要}
删除黑板变量

\SACTitle{语法}
\begin{SACSTX}
UNSETBB ALL|variable ...
\end{SACSTX}

\SACTitle{输入}
\begin{description}
\item [ALL] 删除已定义的全部黑板变量
\item [variable] 删除黑板变量 !variable!
\end{description}

\SACTitle{示例}
一次删除多个黑板变量：
\begin{SACCode}
SAC> unsetbb c1 c2 x
\end{SACCode}

删除所有黑板变量：
\begin{SACCode}
SAC> unsetbb all
\end{SACCode}
