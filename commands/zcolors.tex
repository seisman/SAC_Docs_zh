\SACCMD{zcolors}
\label{cmd:zcolors}

\SACTitle{概要}
控制等值线的颜色显示

\SACTitle{语法}
\begin{SACSTX}
ZCOLORS [ON|OFF] LIST c1 c2 ... cn
\end{SACSTX}

\SACTitle{输入}
\begin{description}
\item [ON] 打开等值线颜色显示开关
\item [OFF] 关闭等值线颜色显示开关
\item [LIST c1 c2 . cn] 设置等值线要使用的颜色列表，每一个颜色对应一条等值线，如果等值线数目多于这个列表长度，则整个列表不断重复
\item [cn] SAC当前颜色表的颜色名
\end{description}

\SACTitle{缺省值}
\begin{SACDFT}
zcolors off list red green blue
\end{SACDFT}
