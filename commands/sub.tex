\SACCMD{sub}
\label{cmd:sub}

\SACTitle{概要}
数据文件的每个数据点减去同一个常数

\SACTitle{语法}
\begin{SACSTX}
SUB  [v1 [v2 ... vn]]
\end{SACSTX}

\SACTitle{输入}
\begin{description}
\item [v1] 从第一个文件中减去的常数
\item [v2] 从第二个文件中减去的常数
\item [vn] 从第n个文件中减去的常数
\end{description}

\SACTitle{缺省值}
\begin{SACDFT}
sub 0.0
\end{SACDFT}

\SACTitle{说明}
参见 \nameref{cmd:add} 的相关说明。

\SACTitle{头段变量改变}
depmin、depmax、depmen
