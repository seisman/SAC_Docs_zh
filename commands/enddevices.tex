\SACCMD{enddevices}
\label{cmd:enddevices}

\SACTitle{概要}
结束某个图像设备

\SACTitle{语法}
\begin{SACSTX}
E!ND!D!EVICES! S!GF!|X!WINDOWS!
\end{SACSTX}

\SACTitle{输入}
\begin{description}
\item [SGF] SAC图形文件设备
\item [XWINDOWS] X Window System图像窗口系统
\end{description}

\SACTitle{说明}
参见命令 \nameref{cmd:begindevices} 的说明。
