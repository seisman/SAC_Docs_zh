\def\cleardoublepage{
    \clearpage

            \c@page
        \else

            \vspace*{\fill}
            \begin{center}
		    保护环境，从阅读电子文档开始！
            \end{center}
            \vspace{\fill}
            \thispagestyle{empty}
            \newpage
            \if@twocolumn
                \hbox{}
                \newpage
            \fi
        \fi
    \fi
}
\makeatother
