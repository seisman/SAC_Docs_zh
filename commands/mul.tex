\SACCMD{mul}
\label{cmd:mul}

\SACTitle{概要}
数据文件的每个数据点乘以同一个常数

\SACTitle{语法}
\begin{SACSTX}
MUL [v1 [v2 ... vn]]
\end{SACSTX}

\SACTitle{输入}
\begin{description}
\item [v1] 乘到第一个文件的常数
\item [v2] 乘到第二个文件的常数
\item [vn] 乘到第n个文件的常数
\end{description}

\SACTitle{缺省值}
\begin{SACDFT}
mul 1.0
\end{SACDFT}

\SACTitle{说明}
参见 \nameref{cmd:add} 的相关说明。

\SACTitle{头段变量改变}
depmin、depmax、depmen
